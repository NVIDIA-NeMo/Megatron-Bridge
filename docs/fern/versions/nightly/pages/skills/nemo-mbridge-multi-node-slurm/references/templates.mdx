# Multi-Node Slurm Templates

## Full Template

```bash
#!/bin/bash
# ==============================================================================
# &lt;MODEL_NAME&gt; &lt;pretrain|sft&gt; — &lt;Framework: MLM | Megatron Bridge&gt;
#
# Default: TP&lt;X&gt; PP&lt;Y&gt; EP&lt;Z&gt;, NNODES=&lt;N&gt; (&lt;N*8&gt; GPUs), MBS=&lt;M&gt;, GBS=&lt;G&gt;
#
# Usage:
#   sbatch &lt;script_name&gt;.sh
# ==============================================================================

#SBATCH --job-name=&lt;job-name&gt;
#SBATCH --nodes=&lt;NNODES&gt;
#SBATCH --ntasks-per-node=1
#SBATCH --gpus-per-node=8
#SBATCH --time=00:30:00
#SBATCH --account=&lt;YOUR_ACCOUNT&gt;
#SBATCH --partition=batch
#SBATCH --output=&lt;SHARED_FS&gt;/logs/&lt;job_name&gt;_%j.log
#SBATCH --exclusive

# ── Container ────────────────────────────────────────────────────────────
CONTAINER_IMAGE="&lt;PATH_TO_YOUR_CONTAINER&gt;.sqsh"
CONTAINER_MOUNTS="&lt;SHARED_FS&gt;:&lt;SHARED_FS&gt;,&lt;PATH_TO_MEGATRON_BRIDGE&gt;:/opt/Megatron-Bridge,&lt;PATH_TO_DATA&gt;:/opt/data"

# ── Paths ────────────────────────────────────────────────────────────────
WORKDIR="/opt/Megatron-Bridge"
LOGDIR="&lt;SHARED_FS&gt;/logs/&lt;logdir_name&gt;"
DATA_PATH="&lt;PATH_TO_PREPROCESSED_DATA&gt;/dclm_01_01_text_document"

# ── Parallelism ──────────────────────────────────────────────────────────
TP=1; PP=1; EP=1

# ── Training ─────────────────────────────────────────────────────────────
MBS=1; GBS=256
SEQ=4096
SEED=1234
TRAIN_ITERS=20

# ── Tokens / Caches ──────────────────────────────────────────────────────
# Provide tokens through the scheduler environment or a chmod 600 secrets file.
# Never hardcode token values in this script or write them to logs.
: "${HF_TOKEN:?Set HF_TOKEN in the secure job environment before submitting}"
export HF_HOME=&lt;SHARED_FS&gt;/HF_HOME
export UV_CACHE_DIR="&lt;SHARED_FS&gt;/uv_cache"
export NEMO_HOME="&lt;SHARED_FS&gt;/cache/nemo"

# ── Build training command ───────────────────────────────────────────────
TRAIN_CMD="
export CUDA_DEVICE_MAX_CONNECTIONS=1 && \
export NVTE_ALLOW_NONDETERMINISTIC_ALGO=1 && \
export PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True && \
export NCCL_NVLS_ENABLE=0 && \
export HF_HOME=$HF_HOME && \
export UV_CACHE_DIR=$UV_CACHE_DIR && \
export NEMO_HOME=$NEMO_HOME && \
wandb login \$WANDB_API_KEY && \
mkdir -p $LOGDIR && \
cd $WORKDIR && \
uv sync && \
&lt;TRAINING_COMMAND_HERE&gt;
"

echo \"======================================\"
echo \"&lt;MODEL_NAME&gt; &lt;Framework&gt; Pretrain\"
echo \"Job: \$SLURM_JOB_ID | Nodes: \$SLURM_JOB_NUM_NODES\"
echo \"TP=\$TP PP=\$PP EP=\$EP MBS=\$MBS GBS=\$GBS\"
echo \"======================================\"

# Phase 1: Single-process uv sync to build/populate the shared cache
srun --mpi=pmix -N 1 --ntasks=1 \
  --container-image="$CONTAINER_IMAGE" \
  --container-mounts="$CONTAINER_MOUNTS" \
  --no-container-mount-home \
  bash -c "cd $WORKDIR && uv sync"

# Phase 2: Full multi-node run (uv sync in TRAIN_CMD is a fast no-op)
srun --mpi=pmix --no-kill \
  --container-image="$CONTAINER_IMAGE" \
  --container-mounts="$CONTAINER_MOUNTS" \
  --no-container-mount-home \
  bash -c "$TRAIN_CMD" 2&gt;&1 | tee "$LOGDIR/&lt;prefix&gt;_${SLURM_JOB_ID}.log"

echo ""
echo "======================================"
echo "Done. Losses:"
echo "======================================"
grep -E "iteration\s+" "$LOGDIR/&lt;prefix&gt;_${SLURM_JOB_ID}.log" | grep -iE "lm loss|reduced_train_loss" | head -25
```
